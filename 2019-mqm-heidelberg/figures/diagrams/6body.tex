\def\angle{15}
\def\tLength{0.5}
\def\tHeight{0.5}
\newcommand{\tamplitude}[1]{
  % define points
  \draw[draw=none]
    (0,0)
      coordinate (#1bottoml)
    ++ (\tLength,0)
      coordinate (#1bottomr)
    (#1bottoml) ++ (90 - \angle:\tHeight)
      coordinate (#1topparticlel)
    (#1bottoml) ++ (90 + \angle:\tHeight)
      coordinate (#1topholel)
    (#1bottomr) ++ (90 - \angle:\tHeight)
      coordinate (#1topparticler)
    (#1bottomr) ++ (90 + \angle:\tHeight)
      coordinate (#1topholer)
  ;

  \draw[] (#1bottoml) -- (#1bottomr);

}
